\begin{document}
	\begin{gather}
		\lim_{x\to0}\frac{\sin(x)}{x}=1\\
		\lim_{x\to0}\frac{\tan(x)}{x}=1\\
		\lim_{x\to0}\frac{1-\cos^2(x)}{x^2}=1\\
		\lim_{x\to0}\frac{1-\cos(x)}{x}=0\\
		\lim_{x\to\infty}\frac{\sin(x)}{x}=0\\
		\sin'(x)=\cos(x)\\
		\cos'(x)=-\sin(x)\\
		\tan'(x)=\frac{1}{\cos^2(x)}=\sec^2(x)\\
		\cot'(x)=-\frac{1}{\sin^2(x)}=\csc^2(x)\\
		\sec'(x)=\sec(x)\tan(x)\\
		\csc'(x)=-\csc(x)\cot(x)\\
	\end{gather}
