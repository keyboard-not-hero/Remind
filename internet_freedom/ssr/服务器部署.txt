1.安装git：[root@www /]# yum install git2.下载shadowsocksr源：[root@www /]# git clone -b manyuser https://github.com/shadowsocksrr/shadowsocksr.git3.初始化git库：[root@www /]# cd ~/shadowsocksr[root@www shadowsocksr]# bash initcfg.sh4.修改配置(修改示例五类Type-Value即可)[root@www shadowsocksr]# vi user-config.json “server_port”:8388,        //端口 “password”:”PASSWD”,     //密码 “protocol”:”origin”,       //协议插件 “obfs”:”plain”,      //混淆插件 “method”:”aes-256-cfb”,    //加密方式</pre>5.默认以配置文件启动ssr(当前目录为多数据库用户，子目录为单用户)[root@www shadowsocksr]# cd shadowsocks                [root@www shadowsocks]# python server.py -d start6.查看端口状态并打开该端口防火墙[root@www shadowsocksr]# firewall-cmd --list-ports[root@www shadowsocksr]# firewall-cmd --add-port=8388/tcp[root@www shadowsocksr]# firewall-cmd --add-port=8388/udp
