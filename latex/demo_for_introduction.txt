7.插入已有图片
\documentclass[UTF8, fontset=ubuntu]{ctexart}
\usepackage{graphicx}
\begin{document}
    \begin{figure}[ht]
        \centering
        \includegraphics[scale=0.4]{snuggle-4713013_640.jpg}
        \caption{这是一张图片}
        \label{fig:snuggle}
    \end{figure}
\end{document}

% \usepackage{graphicx}提供插入图片的包
% \begin{figure}给图片提供浮动空间，ht分别代表位于字体行中间和页顶部
% \centering代表环境内容居中显示。环境代指最近\begin...\end包围
% \includegrapgics[scale=0.4]{index}代表插入图片缩放比例和缩放图片名称
% \caption代表图片注解文字
% \label代表给图片附带可引用标签。引用格式：\ref{fig:snuggle}

8.单行多列图片
\usepackage{graphicx}
\usepackage{subfigure}
\usepackage{float}
\begin{figure}[H]
    \subfigure[奇函数]{
        \includegraphics[scale=0.4]{odd_func.png}
    }   
    \subfigure[偶函数]{
        \includegraphics[scale=0.4]{even_func.png}
    }   
\end{figure}

% usepackage{subfigure}代表行列式排列
% usepackage{float}与\begin{figure}[H]中的H结合，使图片紧贴文字排列，避免图片紧贴一起


9.基础表格
\documentclass[UTF8, fontset=ubuntu]{ctexart}
\begin{document}
    \begin{tabular}{|r r r|}
        \hline
        姓名 & 身高(cm) & 体重(kg)\\
        \hline
        李雷 & 169 & 70\\
        张丽 & 163 & 52\\
        王飞 & 175 & 72\\
        \hline
    \end{tabular}
\end{document}

% \begin{tabular}代表表格环境，{|rrr|}代表表格左右两边被竖线包围，内部含3列，每列都右对齐(r，其余还有l/c分别对应左对齐，居中)
% \hline为横线
% &为列与列之间的分隔

10.自适应宽度表格
\documentclass[UTF8, fontset=ubuntu]{ctexart}
\usepackage{array}
\begin{document}
    \begin{tabular}{|m{2em}|m{2cm}|m{2cm}|}
        \hline
        姓名 & 身高(cm) & 体重(kg)\\
        \hline
        李雷 露西 李莉 & 169 & 70\\
        张丽 & 163 & 52\\
        王飞 & 175 & 72\\
        \hline
    \end{tabular}
\end{document}

% m{2em}限制第一列只占用2列字符宽度，超过该长度拓展到下一行，使之实现自适应。需配合\usepackage{array}使用

11.带标签和标题的表格
\documentclass[UTF8, fontset=ubuntu]{ctexart}
\begin{document}
\begin{table}
    \begin{tabular}{|r r r|}
        \hline
        姓名 & 身高(cm) & 体重(kg)\\
        \hline
        李雷 & 169 & 70\\
        张丽 & 163 & 52\\
        王飞 & 175 & 72\\
        \hline
    \end{tabular}
    \caption{学生信息}
    \label{table:1}
\end{table}
\end{document}

% \begin{table}...\end{table}用于给表格tabular添加额外信息，如caption/label等
% \label{table:1}创建的标签，可以在其他地方使用\ref{table:1}引用
